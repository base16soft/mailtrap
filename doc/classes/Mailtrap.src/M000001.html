  def initialize( host, port, once, msgfile )
    @host = host
    @port = port
    @once = once
    @msgfile = msgfile
    
    File.open( @msgfile, "a" ) do |file|
      file.puts "\n* Mailtrap started at #{@host}:#{port}\n"
    end
    
    service = TCPServer.new( @host, @port )
    accept( service )
  end
